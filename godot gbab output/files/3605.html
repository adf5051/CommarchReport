class PropertySelector : public ConfirmationDialog {
	OBJ_TYPE(PropertySelector,ConfirmationDialog )


	LineEdit *search_box;
	Tree *search_options;

	void _update_search();

	void _sbox_input(const InputEvent& p_ie);

	void _confirmed();

#endif // PROPERTYSELECTOR_H
